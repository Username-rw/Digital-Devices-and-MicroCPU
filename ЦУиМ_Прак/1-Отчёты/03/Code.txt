#include <8051.h>

void lcd_command(unsigned char cmd);
void lcd_data(unsigned char data);
void lcd_clear();
void lcd_set_cursor(unsigned char row, unsigned char col);

void lcd_command(unsigned char cmd) {
    P0 = cmd;      
    P2 = 0x1;       
    P2 = 0x0;       
}

void lcd_data(unsigned char data) {
    P0 = data;      
    P2 = 0x3;       
    P2 = 0x2;       
}

void lcd_clear() {
    lcd_command(0x01); 
}

void lcd_set_cursor(unsigned char row, unsigned char col) {
    if (row == 0) {
        lcd_command(0x80 + col); 
    } else {
        lcd_command(0xC0 + col); 
    }
}

void main() {
    unsigned char str[] = "WORLD"; 
    unsigned char len = 5; 
    unsigned char pos, i;

    lcd_command(0x38); 
    lcd_command(0x0C); 
    lcd_command(0x06); 
    lcd_clear();

    while (1) {
        for (pos = 0; pos < 8 + len; pos++) { 
            lcd_clear(); 
            if (pos >= 8) {
                lcd_set_cursor(0, pos - 8); 
                for (i = 0; i < len; i++) {
                    lcd_data(str[i]);
                }
            } else {
                lcd_set_cursor(1, pos); 
                for (i = 0; i < len; i++) {
                    lcd_data(str[i]);
                }
            }
            for (i = 0; i < 2; i++); 
        }
    }
}
