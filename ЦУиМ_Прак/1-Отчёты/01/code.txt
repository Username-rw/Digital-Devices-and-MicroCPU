#include <8051.h> //подключение заголовочного файла
void main() //главная функция, точка входа в программу
{
int i;
char xdata *ptr; //указатель на адрес ячейки памяти внешнего ОЗУ
char test, nabor;
nabor = 0FFh; //тестовый набор
ptr = (char xdata *) 1200h; //начальный адрес внешней памяти
for(i=0; i<1024;i++) //проверка 1024 ячеек внешней памяти
{
	*ptr=nabor; //в каждую ячейку памяти посылается значение тестового набора
	test=*ptr; //переменная test принимает значение содержимого ячейки памяти
	if(test!=nabor)
	{
		break; //в случае ошибки цикл завершается
	}
	ptr++;
	} 
}
